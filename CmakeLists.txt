cmake_minimum_required(VERSION 3.20)

project(DCmcuos
    VERSION 0.0.0.1
    LANGUAGES CXX
    DESCRIPTION "DC MCU Pivot - A multi-task scheduling framework for microcontrollers"
)

set(CMAKE_C_STANDARD 99)
set(CMAKE_CXX_STANDARD 23)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# ----------------------------------
# Build Options
# ----------------------------------
option(DC_MCU_PIVOT_BUILD_TESTS "Build test cases" ON)
option(DC_MCU_PIVOT_BUILD_EXAMPLES "Build examples" OFF)
option(DC_MCU_PIVOT_INSTALL "Install library" ON)
option(DC_MCU_PIVOT_INSTALL_DEPLOY "Install with deploying dependencies" OFF)
option(DC_MCU_PIVOT_INSTALL_MSVC_RUNTIME "Install MSVC runtime" OFF)

# ----------------------------------
# CMake Settings
# ----------------------------------
if(MSVC)
    set(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} /manifest:no")
    set(CMAKE_MODULE_LINKER_FLAGS "${CMAKE_MODULE_LINKER_FLAGS} /manifest:no")
    set(CMAKE_SHARED_LINKER_FLAGS "${CMAKE_SHARED_LINKER_FLAGS} /manifest:no")
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} /utf-8")

    if(NOT DEFINED CMAKE_DEBUG_POSTFIX)
        set(CMAKE_DEBUG_POSTFIX "d")
    endif()
elseif(MINGW)
    set(CMAKE_STATIC_LIBRARY_PREFIX "")
    set(CMAKE_SHARED_LIBRARY_PREFIX "")
else()
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -fPIC")
endif()

if(DC_MCU_PIVOT_INSTALL)
    include(GNUInstallDirs)
    include(CMakePackageConfigHelpers)
endif()

# ----------------------------------
# Project Variables
# ----------------------------------
set(DC_MCU_PIVOT_VERSION ${PROJECT_VERSION})
set(DC_MCU_PIVOT_INSTALL_NAME ${PROJECT_NAME})
set(DC_MCU_PIVOT_SOURCE_DIR ${CMAKE_CURRENT_SOURCE_DIR})

# ----------------------------------
# Find basic dependencies
# ----------------------------------
find_package(qmsetup REQUIRED)
find_package(etl CONFIG REQUIRED)

qm_import(Filesystem)
qm_init_directories()

# ----------------------------------
# Add source modules
# ----------------------------------
include("${DC_MCU_PIVOT_SOURCE_DIR}/cmake/BuildAPI.cmake")
add_subdirectory(API)
add_subdirectory(main)
add_subdirectory(RTOS)


# ----------------------------------
# Compile configuration
# ----------------------------------
add_compile_options(
    $<$<COMPILE_LANGUAGE:CXX>:-fno-exceptions>
    $<$<COMPILE_LANGUAGE:CXX>:-fno-rtti>
    -Wall
    -Wextra
)

# ----------------------------------
# Release Mode Optimization
# ----------------------------------
# 添加优化级别选项
option(DC_MCU_PIVOT_OPTIMIZE_SIZE "Optimize for size instead of speed" ON)

# 针对不同编译器和平台设置优化选项
if(CMAKE_BUILD_TYPE STREQUAL "Release" OR CMAKE_BUILD_TYPE STREQUAL "MinSizeRel")
    message(STATUS "Configuring for Release build with optimizations")
    
    if(MSVC)
        # MSVC 优化选项
        add_compile_options(
            /O2     # 最大化速度优化
            /Ob2    # 激进的内联展开
            /GF     # 消除重复字符串
            /Gy     # 函数级链接
        )
        
        if(DC_MCU_PIVOT_OPTIMIZE_SIZE)
            add_compile_options(/Os)  # 优先优化代码大小
        else()
            add_compile_options(/Ot)  # 优先优化执行速度
        endif()
        
        # 链接时优化(LTO)
        if(CMAKE_VERSION VERSION_GREATER_EQUAL 3.18)
            include(CheckIPOSupported)
            check_ipo_supported(RESULT ipo_supported OUTPUT output)
            if(ipo_supported)
                set(CMAKE_INTERPROCEDURAL_OPTIMIZATION ON)
                message(STATUS "Enabling Link Time Optimization (LTO)")
            endif()
        else()
            add_compile_options(/GL)
            add_link_options(/LTCG)
        endif()
        
    else()
        # GCC/Clang 优化选项
        if(DC_MCU_PIVOT_OPTIMIZE_SIZE)
            add_compile_options(-Os)  # 优化代码大小
        else()
            add_compile_options(-O3)  # 最大化优化级别
        endif()
        
        # 通用优化选项
        add_compile_options(
            -ffunction-sections  # 将每个函数放在单独段中
            -fdata-sections      # 将每个数据放在单独段中
            -flto               # 链接时优化
        )
        
        # 链接选项
        add_link_options(
            -Wl,--gc-sections   # 移除未使用代码段
            -flto
        )
        
        # 特定于ARM的优化
        if(CMAKE_SYSTEM_PROCESSOR MATCHES "arm")
            add_compile_options(
                -mcpu=cortex-m4  # 根据实际MCU调整
                -mfloat-abi=hard
                -mfpu=fpv4-sp-d16
            )
        endif()
    endif()
    
    # 额外优化选项
    add_compile_options(
        -fomit-frame-pointer  # 省略帧指针
        -fno-stack-protector   # 禁用栈保护
    )
    
    # 移除调试符号
    if(NOT DC_MCU_PIVOT_DEBUG_SYMBOLS)
        if(MSVC)
            add_compile_options(/Z7)  # 调试信息嵌入在OBJ中
        else()
            add_compile_options(-g0)  # 完全移除调试信息
        endif()
    endif()
endif()
